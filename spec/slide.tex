	\subsection{Les couches d'abstraction}

		\begin{frame}
			\frametitle{La couche physique}
			\framesubtitle{Les bits}
			\begin{block}{Différents bits}
				\begin{itemize}
					\item Le bit de start
					\item Le bit "haut" logique
					\item Le bit "bas" logique
					\item Le bit de stop
				\end{itemize}
			\end{block}
			\begin{block}{Définition des états logiques}
				\includegraphics[width=4cm]{image/chronogramme-2bits-new.png}
			\end{block}
		\end{frame}

		\begin{frame}
			\frametitle{La couche physique}
			\framesubtitle{La trame}
			\begin{block}{Trame de 10 bits}
				\begin{itemize}
					\item Le bit de start
					\item 8 bits de data
					\item le bit de parité
					\item le bit de stop
				\end{itemize}
			\end{block}
			\begin{block}{schéma d'un paquet}
				\includegraphics[width=8cm]{image/chronogramme-octet.png}
			\end{block}
		\end{frame}

		\begin{frame}
			\frametitle{La couche de transport}
			\begin{block}{Au moins 4 octets}
				\begin{itemize}
					\item L'adresse source du paquet (un octet)
					\item L'adresse de destination du paquet (un octet)
					\item La taille et le checksum (un octet)
					\begin{itemize}
						\item La taille sur 4 bit
						\item le checksum sur 4 bit
					\end{itemize}
					\item Les datas (entre 1 et 16 octets)
				\end{itemize}
			\end{block}
		\end{frame}

		\begin{frame}
			\frametitle{La couche applicative}
			\begin{block}{Fonction de haut niveau}
				\begin{itemize}
					\item Fonction d'envoie d'un paquet prenant en paramètre :
					\begin{itemize}
						\item L'adresse de destination
						\item La taille des datas
						\item Les datas
					\end{itemize}
					\item Fonction de réception d'un paquet (aller interroger le buffer)
					\begin{itemize}
						\item Adresse source du paquet
						\item Taille des datas
						\item Datas
					\end{itemize}
					\item Fonction de traitement des erreurs
					\begin{itemize}
						\item Problème de parité
						\item Erreur du checksum
						\item Buffer plein
					\end{itemize}
				\end{itemize}
			\end{block}
		\end{frame}

	\subsection{Les problèmes rencontré}


		\begin{frame}
			\frametitle{Le compilateur}
			\begin{alertblock}{Pas de print dans la boucle}
				while(1)\{\newline
				if( reception == 1 )\{\newline
				reception = 0;\newline
				print("octet recu");\}\}
			\end{alertblock}
			\begin{exampleblock}{Print dans la boucle}
				while(1)\{\newline
				print("coucou");\newline
				if( reception == 1 )\{\newline
				reception = 0;\newline
				print("octet recu");\}\}
			\end{exampleblock}
		\end{frame}

		\begin{frame}
			\frametitle{La désynchronisation}
			\begin{alertblock}{Désynchronisation des bits lors de la réception d'un octet}
				Solution : modifier le bit de start pour qu'il soit unique
			\end{alertblock}
			\begin{alertblock}{Désynchronisation des octets lors de la réception d'un paquet}
				Solution : créer un octet de start de paquet (modification de l'octet d'adresse pour qu'il commence toujours par la même chose)
			\end{alertblock}
		\end{frame}

	\subsection{Réception non bloquante}
		
		\begin{frame}
			\frametitle{Ne pas bloquer le programme principal}
			\begin{itemize}
				\item Interruption sur front montant qui déclenche le timer
				\item Interruption toute les millisecondes sur le timer (s'il est déclenché)
				\item Analyse, 5 millisecondes après l'interruption sur front montant, du signal
				\item Une fois un octet reçu entièrement, passage de l'octet pour analyse de celui-ci par la couche supérieur
				\item Création d'un paquet stocké dans un buffer
			\end{itemize}
		\end{frame}
